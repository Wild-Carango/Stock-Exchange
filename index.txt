let filters = {position:'all', minval:0, clustersOnly:false};
let sortCol = 9, sortDir = -1;

function getBadge(title) {
  const t = title.toLowerCase();
  if (t.includes('chief executive')||t.includes('ceo')) return `<span class="badge badge-ceo">CEO</span>${title}`;
  if (t.includes('chief financial')||t.includes('cfo')) return `<span class="badge badge-cfo">CFO</span>${title}`;
  if (t.includes('director'))  return `<span class="badge badge-dir">DIR</span>${title}`;
  return `<span class="badge badge-other">INS</span>${title}`;
}

function renderTable(data) {
  const tbody = document.getElementById('table-body');
  if (!data.length) {
    tbody.innerHTML = "<tr><td colspan='11' style='text-align:center;padding:40px;color:#999'>No results match your filters</td></tr>";
    document.getElementById('results-count').textContent = '0 purchases shown';
    document.getElementById('stat-count').textContent = '0';
    return;
  }
  tbody.innerHTML = data.map(p => {
    const cl = CLUSTER_COS.has(p.company) ? ' <span class="cluster-badge">CLUSTER</span>' : '';
    return `<tr>
      <td>${p.filed}</td><td>${p.date}</td>
      <td><strong>${p.company}</strong>${cl}</td>
      <td class="ticker">${p.ticker}</td>
      <td>${p.insider}</td>
      <td class="pos">${getBadge(p.title)}</td>
      <td>${p.security}</td>
      <td class="num">${p.shares.toLocaleString()}</td>
      <td class="num">${p.price}</td>
      <td class="tot">${p.total}</td>
      <td class="num">${p.after.toLocaleString()}</td>
    </tr>`;
  }).join('');
  document.getElementById('results-count').textContent = `${data.length} purchases shown`;
  document.getElementById('stat-count').textContent = data.length;
}

function applyFilters() {
  const search = document.getElementById('search-box').value.toLowerCase();
  let filtered = DATA.filter(p => {
    if (filters.position !== 'all') {
      const t = p.title.toLowerCase();
      if (filters.position==='ceo' && !t.includes('chief executive') && !t.includes('ceo')) return false;
      if (filters.position==='cfo' && !t.includes('chief financial') && !t.includes('cfo')) return false;
      if (filters.position==='director' && !t.includes('director')) return false;
      if (filters.position==='officer' && !t.includes('officer') && !t.includes('ceo') && !t.includes('cfo') && !t.includes('coo')) return false;
    }
    if (p.totalRaw < filters.minval) return false;
    if (filters.clustersOnly && !CLUSTER_COS.has(p.company)) return false;
    if (search && !p.company.toLowerCase().includes(search) && !p.insider.toLowerCase().includes(search) && !p.ticker.toLowerCase().includes(search)) return false;
    return true;
  });
  filtered.sort((a,b) => {
    const av=[a.filed,a.date,a.company,a.ticker,a.insider,a.title,a.security,a.shares,a.price,a.totalRaw,a.after][sortCol];
    const bv=[b.filed,b.date,b.company,b.ticker,b.insider,b.title,b.security,b.shares,b.price,b.totalRaw,b.after][sortCol];
    return av<bv ? sortDir : av>bv ? -sortDir : 0;
  });
  renderTable(filtered);
}

function setFilter(type, val, btn) {
  filters[type] = val;
  btn.closest('.filters').querySelectorAll('.filter-btn').forEach(b => {
    if (b.getAttribute('onclick') && b.getAttribute('onclick').includes(`'${type}'`)) b.classList.remove('active');
  });
  btn.classList.add('active');
  applyFilters();
}

function toggleClusters(btn) {
  filters.clustersOnly = !filters.clustersOnly;
  btn.textContent = filters.clustersOnly ? 'On' : 'Off';
  btn.classList.toggle('active', filters.clustersOnly);
  applyFilters();
}

function sortTable(col) {
  sortDir = sortCol===col ? -sortDir : -1;
  sortCol = col;
  document.querySelectorAll('thead th').forEach((th,i) => {
    th.classList.remove('sorted-asc','sorted-desc');
    if (i===sortCol) th.classList.add(sortDir===-1?'sorted-desc':'sorted-asc');
  });
  applyFilters();
}

function showTab(name) {
  document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));
  document.querySelectorAll('.tab-content').forEach(t=>t.classList.remove('active'));
  const idx = ['all','watchlist','clusters'].indexOf(name);
  document.querySelectorAll('.tab')[idx].classList.add('active');
  document.getElementById('tab-'+name).classList.add('active');
}
